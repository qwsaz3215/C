Recursion

#include<stdlib.h>

int fac(int);
int main(void)
{
	int n;
	printf("Please enter the integer:");
	scanf("%d",&n);
	printf("fac(n)=%d\n", fac(n));
	
	system("pause");
	return 0;
}

int fac(int n)
{
	if(n>0)
		return (n*fac(n-1));
		else
			return 1;
}

# D函數設計題

fib_for


#include<stdio.h>
#include<stdlib.h>

int fib(int);
int main(void)
{
	int a,b;
	printf("Please enter the integer:");
	scanf("%d",&a);
	
	for(b=1;b<=a;b++)
		printf("fib(%d)=%d\n",b,fib(b));
		
		system("pause");
		return 0;
}

int fib(int b)
{
	if(b==1 || b==2)
		return 1;
	else
		return(fib(b-1)+fib(b-2));
}
